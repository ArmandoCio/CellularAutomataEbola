int createTable(int twoDtable[][], int tablerow, int tablecol)
{
    for (int i = 0; i < tablerow; i++)
    {
        for (int j = 0; j < tablecol; j++)
        {
            twoDtable[i][j] = i + j 
        }
    }

    return twoDtable

}
